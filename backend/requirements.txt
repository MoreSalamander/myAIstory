# my-AI-story backend dependencies.
#
# Core (phases 1-2): the verify/ package depends on nothing but the schema
# layer; pydantic provides the typed contracts and pytest runs the gates.
# The Ollama client uses only the stdlib (urllib), so no HTTP library here.
pydantic>=2.6
pytest>=8.0

# Phase 3 — local TTS (optional at runtime; the pipeline is text-only without
# an engine). piper-tts is the v1 backend; voice models are fetched separately
# via `python -m piper.download_voices <name> --data-dir data/voices`.
# certifi gives that downloader a CA bundle on framework Python builds.
piper-tts>=1.4
certifi

# Higher-fidelity local TTS (optional): Kokoro-82M via ONNX Runtime — markedly
# more natural than Piper, still fully offline. Model files are fetched once
# into data/voices_kokoro/ (kokoro-v1.0.onnx + voices-v1.0.bin) from the
# kokoro-onnx release. Selected per run with --kokoro / engine="kokoro".
kokoro-onnx>=0.4

# Voice cloning (optional, heaviest backend): Chatterbox (Resemble AI). Unlike
# Piper/Kokoro's fixed timbres, this matches a SHORT reference clip you supply
# (data/voice_refs/<id>.wav) — so you can cast a specific voice, with an
# exaggeration knob for delivery intensity. Pulls in PyTorch (+ transformers,
# gradio, etc.) and downloads the model once into the HF cache. Install only
# when you want cloning (selected with --clone / engine="clone"); left commented
# so a plain `pip install -r` doesn't force a multi-GB download on every setup.
# chatterbox-tts>=0.1.7

# Phase 4 — web UI. FastAPI serves the frontend and streams run events as
# NDJSON; uvicorn is the ASGI server; httpx backs the test client.
fastapi>=0.115
uvicorn[standard]>=0.30
httpx
